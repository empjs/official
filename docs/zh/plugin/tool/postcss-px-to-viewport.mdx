# postcss-px-to-viewport 插件

## 快速开始

### 安装插件
import { PackageManagerTabs } from '@theme';

<PackageManagerTabs command="add @empjs/postcss-px-to-viewport -D" />

### 使用插件
```ts title="emp-config.js"
import {defineConfig} from '@empjs/cli'
export default defineConfig(store => {
  return {
    plugins,
    css: {
      postcss: [
        [
              store.importResolve('@empjs/postcss-px-to-viewport', import.meta.url),
              {
                unitToConvert: 'px',
                viewportWidth: 320,
                viewportHeight: 568, 
                unitPrecision: 5,
                viewportUnit: 'vw',
                fontViewportUnit: 'vw',
                selectorBlackList: [],
                propList: ['*'],
                minPixelValue: 1,
                mediaQuery: false,
                replace: true,
                landscape: false,
                landscapeUnit: 'vw',
                landscapeWidth: 568,
              },
        ],
      ],
    },
  }
})
```
