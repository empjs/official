import {defineConfig} from '@empjs/cli'
export default defineConfig(store => {
    // 方法一
    store.chain.module.rule('javascript').test(/\.js$/)
  return {
    chain(chain){
        // 方法二
        chain.module.rule('javascript').test(/\.js$/)
    },
  }
})
